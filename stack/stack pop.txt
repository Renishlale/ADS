 #include <stdio.h>

int main() {
    int size, n, top = -1;

    printf("Enter the size of the stack: ");
    scanf("%d", &size);
    int stack[size];

    printf("Enter the number of elements to be entered: ");
    scanf("%d", &n);

    if (n > size) {
        printf("\n%d number of elements cannot be pushed\n", n);
    } else {
        for (int i = 0; i < n; i++) {
            int value;
            printf("\nEnter the value of element %d: ", i + 1);
            scanf("%d", &value);
            top++;
            stack[top] = value;
            printf("%d pushed into the stack\n", value);
        }
    }

    if (top == -1) {
        printf("Stack is empty, cannot pop element\n");
    } else {
        printf("\nElement popped: %d\n", stack[top]);
        top--;
    }

    printf("\nStack after popping:\n");
    printf("-------------Stack--------------\n");

    if (top == -1) {
        printf("Stack is empty\n");
    } else {
        for (int i = top; i >= 0; i--) {
            printf("%d\n", stack[i]);
        }
    }

    return 0;
}
