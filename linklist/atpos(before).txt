  
#include <stdio.h>
#include<stdlib.h>

struct node
{
    int data;
    struct node*next;
};


int insert_pos(int value,struct node**head,int key)
{
      
    
    struct node*temp=*head; 
   while(temp!=NULL)
   {
       if(temp->data==key)
       {
            struct node*newnode;
            newnode=malloc(sizeof(struct node));
            newnode->data=value;
            newnode->next=temp->next;
            temp->next=newnode;
            return 0;
       }
       temp=temp->next;
   }
   return 0;
}




int insert_beg (int value,struct node**head)
{
    struct node*newnode;
    newnode=malloc(sizeof(struct node));
    newnode->data=value;
    newnode->next=*head;
    *head=newnode;
    return 0;
}

        newnode->data=value;

       struct node*temp=*head;

    return 0;
}
